package com.example.switchtoggleapp;

import android.os.Bundle;
import android.widget.*;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    Switch switch1;
    ToggleButton toggle1;
    Button btnCheck;
    TextView result;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        switch1 = findViewById(R.id.switch1);
        toggle1 = findViewById(R.id.toggle1);
        btnCheck = findViewById(R.id.btnCheck);
        result = findViewById(R.id.result);

        btnCheck.setOnClickListener(v -> {

            StringBuilder status = new StringBuilder();

            // Switch state
            if (switch1.isChecked()) {
                status.append("Switch is ON\n");
                Toast.makeText(this, "Switch ON", Toast.LENGTH_SHORT).show();
            } else {
                status.append("Switch is OFF\n");
                Toast.makeText(this, "Switch OFF", Toast.LENGTH_SHORT).show();
            }

            // Toggle state
            if (toggle1.isChecked()) {
                status.append("Toggle Button is ON");
                Toast.makeText(this, "Toggle ON", Toast.LENGTH_SHORT).show();
            } else {
                status.append("Toggle Button is OFF");
                Toast.makeText(this, "Toggle OFF", Toast.LENGTH_SHORT).show();
            }

            result.setText(status.toString());
        });
    }
}
