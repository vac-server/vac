main.java


package com.example.studentapp;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.*;

import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    EditText name, surname, studentClass, marks;
    RadioGroup genderGroup;
    CheckBox sports, music, reading;
    Button submit;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        name = findViewById(R.id.etName);
        surname = findViewById(R.id.etSurname);
        studentClass = findViewById(R.id.etClass);
        marks = findViewById(R.id.etMarks);

        genderGroup = findViewById(R.id.rgGender);

        sports = findViewById(R.id.cbSports);
        music = findViewById(R.id.cbMusic);
        reading = findViewById(R.id.cbReading);

        submit = findViewById(R.id.btnSubmit);

        submit.setOnClickListener(v -> {

            String n = name.getText().toString();
            String s = surname.getText().toString();
            String c = studentClass.getText().toString();
            String m = marks.getText().toString();

            int selectedId = genderGroup.getCheckedRadioButtonId();
            String gender = "";
            if (selectedId != -1) {
                RadioButton genderBtn = findViewById(selectedId);
                gender = genderBtn.getText().toString();
            }

            String hobbies = "";
            if (sports.isChecked()) hobbies += "Sports ";
            if (music.isChecked()) hobbies += "Music ";
            if (reading.isChecked()) hobbies += "Reading ";

            Intent i = new Intent(MainActivity.this, SecondActivity.class);
            i.putExtra("name", n);
            i.putExtra("surname", s);
            i.putExtra("class", c);
            i.putExtra("gender", gender);
            i.putExtra("hobbies", hobbies);
            i.putExtra("marks", m);

            startActivity(i);
        });
    }
}

package com.example.studentapp;

import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class SecondActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        TextView name = findViewById(R.id.tvName);
        TextView surname = findViewById(R.id.tvSurname);
        TextView studentClass = findViewById(R.id.tvClass);
        TextView gender = findViewById(R.id.tvGender);
        TextView hobbies = findViewById(R.id.tvHobbies);
        TextView marks = findViewById(R.id.tvMarks);

        name.setText(getIntent().getStringExtra("name"));
        surname.setText(getIntent().getStringExtra("surname"));
        studentClass.setText(getIntent().getStringExtra("class"));
        gender.setText(getIntent().getStringExtra("gender"));
        hobbies.setText(getIntent().getStringExtra("hobbies"));
        marks.setText(getIntent().getStringExtra("marks"));
    }
}
